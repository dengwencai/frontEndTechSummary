
```js
window.addEventListener("resize", function () {
  if ([0, 180].includes(window.orientation)) {
    console.log("竖屏");
  }
  if ([90, -90].includes(window.orientation)) {
    console.log("横屏");
  }
});
```

```css
@media screen and (orientation: portrait) {
  /* 竖屏样式代码 */
}

@media screen and (orientation: landscape) {
  /* 横屏样式代码 */
}
```
